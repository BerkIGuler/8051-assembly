ORG 0000h
	MOV R2, #0; NUMBER
	MOV R3, #6; COUNTER
	MOV R1, #40h; ADRESS
AGAIN:	CLR A
	MOV DPTR, #1000h
	MOVC A, @A+DPTR
	
COMPARE:CJNE A, 2, NEXT 
	MOV @R1, A
	INC R1

NEXT:	INC DPTR
	CLR A
	MOVC A, @A+DPTR
	DJNZ R3, COMPARE
	INC R2
	MOV R3, #6
	CJNE R2, #0, AGAIN
	LJMP EXIT

	ORG 1000h
ARRAY:	DB 40, 80, 120, 160, 200, 240
EXIT:	END
